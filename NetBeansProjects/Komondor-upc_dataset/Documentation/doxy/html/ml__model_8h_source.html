135  int ComputeIndividualConfiguration(int arm_ix, double reward, Logger &agent_logger, double sim_time) {
136  int new_action(0);
137  switch(learning_mechanism) {
138  /* MULTI_ARMED_BANDITS */
139  case MULTI_ARMED_BANDITS: {
140  // Update the reward of the last played configuration
141  mab_agent.UpdateArmStatistics(arm_ix, reward);
142  // Select a new action according to the updated information
143  new_action = mab_agent.SelectNewAction();
144  break;
145  }
146  }
