99  void UpdateArmStatistics(int action_ix, double reward){
100 
101  if(action_ix >= 0) { // Avoid indexing errors
102  // Update the reward for the chosen arm
103  reward_per_arm[action_ix] = reward;
104  // Update the times the chosen arm has been selected
105  times_arm_has_been_selected[action_ix] += 1;
106  // Update the cumulative reward for the chosen arm
107  cumulative_reward_per_arm[action_ix] += reward;
108  // Update the average reward for the chosen arm
109  average_reward_per_arm[action_ix] = cumulative_reward_per_arm[action_ix] /
110  times_arm_has_been_selected[action_ix];
111  // Update the estimated reward per arm
112  estimated_reward_per_arm[action_ix] = ((estimated_reward_per_arm[action_ix]
113  * times_arm_has_been_selected[action_ix])
114  + reward) / (times_arm_has_been_selected[action_ix] + 2);
115  } else {
116  printf("[MAB] ERROR: The action ix (%d) is not correct!\n", action_ix);
117  exit(EXIT_FAILURE);
118  }
119  }

125  int SelectNewAction() {
126  int action_ix;
127  // Select an action according to the chosen strategy: TODO improve this part (now it is hardcoded)
128  //action_selection_strategy = STRATEGY_EGREEDY;
129  switch(action_selection_strategy) {
130  /*
131  * epsilon-greedy strategy:
132  */
133  case STRATEGY_EGREEDY:{
134  // Update epsilon
135  epsilon = initial_epsilon / sqrt( (double) num_iterations);
136  // Pick an action according to e-greedy
137  action_ix = PickArmEgreedy(num_actions, average_reward_per_arm, epsilon);
138  // Increase the number of iterations
139  num_iterations ++;
140  break;
141  }
142  /*
143  * Thompson sampling strategy:
144  */
145  case STRATEGY_THOMPSON_SAMPLING:{
146  // Pick an action according to Thompson sampling
147  action_ix = PickArmThompsonSampling(num_actions,
148  estimated_reward_per_arm, times_arm_has_been_selected);
149  // Increase the number of iterations
150  num_iterations ++;
151  break;
152  }
153  default:{
154  printf("[MAB] ERROR: '%d' is not a correct action-selection strategy!\n", action_selection_strategy);
155  PrintAvailableActionSelectionStrategies();
156  exit(EXIT_FAILURE);
157  }
158  }
159  return action_ix;
160  }
