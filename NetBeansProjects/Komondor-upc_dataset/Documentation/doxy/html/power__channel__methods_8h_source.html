69 double ConvertPower(int conversion_type, double power_magnitude_in){
70 
71  double converted_power (0);
72 
73  switch(conversion_type){
74  // pW to dBm
75  case PW_TO_DBM:{
76  converted_power = 10 * log10(power_magnitude_in * pow(10,-9));
77  break;
78  }
79  // dBm to pW
80  case DBM_TO_PW:{
81  converted_power = pow(10,(power_magnitude_in + 90)/10);
82  break;
83  }
84  // mW to dBm
85  case MW_TO_DBM:{
86  converted_power = 10 * log10(power_magnitude_in * pow(10,-6));
87  break;
88  }
89  // dBm to mW (dB to linear)
90  case DBM_TO_MW:
91  case DB_TO_LINEAR:
92  case DBW_TO_W: {
93  converted_power = pow(10,power_magnitude_in/10);
94  break;
95  }
96  // W to dBW
97  case W_TO_DBW:
98  case LINEAR_TO_DB: {
99  converted_power = 10 * log10(power_magnitude_in);
100  break;
101  }
102  default:{
103  printf("Power conversion type not found!\n");
104  break;
105  }
106  }
107  return converted_power;
108 }

135 double ComputePowerReceived(double distance, double tx_power, double tx_gain, double rx_gain,
136  double central_frequency, int path_loss_model) {
137 
138 // printf(" - distance = %f\n", distance);
139 // printf(" - tx_power = %f\n", ConvertPower(PW_TO_DBM,tx_power));
140 // printf(" - tx_gain = %f\n", tx_gain);
141 // printf(" - rx_gain = %f\n", rx_gain);
142 // printf(" - central_frequency = %f\n", central_frequency);
143 // printf(" - path_loss_model = %d\n", path_loss_model);
144 
145  double tx_power_dbm (ConvertPower(PW_TO_DBM, tx_power));
146  double tx_gain_db (ConvertPower(LINEAR_TO_DB, tx_gain));
147  double rx_gain_db (ConvertPower(LINEAR_TO_DB, rx_gain));
148  double pw_received_dbm;
149  double wavelength ((double) SPEED_LIGHT/central_frequency);
150  double loss;
151 
152  double pw_received; // Power received [pW]
153 
154  switch(path_loss_model){
155  // Free space - Calculator: https://www.pasternack.com/t-calculator-fspl.aspx (UNITS ARE NOT IN SI!)
156  case PATH_LOSS_LFS:{
157 
158  pw_received = tx_power * tx_gain * rx_gain * pow(((double) SPEED_LIGHT/(4*M_PI*distance*central_frequency)),2);
159 
160  break;
161  }
162  // Okumura-Hata model - Urban areas
163  case PATH_LOSS_OKUMURA_HATA:{
164  double tx_heigth (10); // Transmitter height [m]
165  double rx_heigth (10); // Receiver height [m]
166  double path_loss_A (69.55 + 26.16 * log10(3*pow(10,8)/wavelength) - 13.82 * log10(tx_heigth));
167  double path_loss_B (44.9 - 6.55 * log10(tx_heigth));
168  double path_loss_E (3.2 * pow(log10(11.7554 * rx_heigth),2) - 4.97);
169  double path_loss (path_loss_A + path_loss_B * log10(distance/1000) - path_loss_E);
170  pw_received_dbm = tx_power_dbm + tx_gain_db + rx_gain_db - path_loss;
171  pw_received = ConvertPower(DBM_TO_PW, pw_received_dbm);
172  break;
173  }
174  // Indoor model (could suite an apartments building scenario)
175  case PATH_LOSS_INDOOR: {
176  double path_loss_factor (5);
177  double shadowing (9.5);
178  double obstacles (30);
179  double walls_frequency (5); // One wall each 5 meters on average
180  double shadowing_at_wlan ((((double) rand())/RAND_MAX)*shadowing);
181  double obstacles_at_wlan ((((double) rand())/RAND_MAX)*obstacles);
182  double alpha (4.4); // Propagation model
183  double path_loss (path_loss_factor + 10*alpha*log10(distance) + shadowing_at_wlan +
184  (distance/walls_frequency)*obstacles_at_wlan);
185  pw_received_dbm = tx_power_dbm + tx_gain_db - path_loss; // Power in dBm
186  pw_received = ConvertPower(DBM_TO_PW, pw_received_dbm);
187  break;
188  }
189  // Indoor model without variability
190  case PATH_LOSS_INDOOR_2: {
191  double path_loss_factor (5);
192  double shadowing (9.5);
193  double obstacles (30);
194  double walls_frequency (5); // One wall each 5 meters on average
195  double shadowing_at_wlan (1/2*shadowing);
196  double obstacles_at_wlan (1/2*obstacles);
197  double alpha (4.4); // Propagation model
198  double path_loss (path_loss_factor + 10*alpha*log10(distance) + shadowing_at_wlan +
199  (distance/walls_frequency)*obstacles_at_wlan);
200  pw_received_dbm = tx_power_dbm + tx_gain_db - path_loss; // Power in dBm
201  pw_received = ConvertPower(DBM_TO_PW, pw_received_dbm);
202  break;
203  }
204 
205  // Residential - 5 dB/wall and 18.3 dB per floor, and 4 dB shadow
206  // Retrieved from: https://mentor.ieee.org/802.11/dcn/14/11-14-0882-04-00ax-tgax-channel-model-document.docx
207  // IEEE 802.11ax uses the TGn channel B path loss model for performance evaluation of simulation scenario #1
208  // with extra indoor wall and floor penetration loss.
209  case PATH_LOSS_SCENARIO_1_TGax: {
210  int n_walls(10); // Wall frequency (n_walls walls each m)
211  int n_floors(3); // Floor frequency (n_floors floors each m)
212  int L_iw(5); // Penetration for a single wall (dB)
213  double min_d(distance);
214  if (distance > 5) { min_d = 5; }
215  double central_frequency_ghz(central_frequency / pow(10,9));
216  double LFS (40.05 + 20*log10(central_frequency_ghz/2.4) + 20*log10(min_d) +
217  18.3*pow((distance/n_floors),(((distance/n_floors)+2)/((distance/n_floors)+1))
218  - 0.46) + L_iw*(distance/n_walls));
219  double d_BP (5); // Break-point distance (m)
220  if (distance >= d_BP) {
221  loss = LFS + 35*log10(distance/double(5));
222  } else {
223  loss = LFS;
224  }
225  pw_received_dbm = tx_power_dbm + tx_gain_db + rx_gain_db - loss;
226  pw_received = ConvertPower(DBM_TO_PW, pw_received_dbm);
227  break;
228  }
229 
230  // Enterprise - 5 dB/wall and 18.3 dB per floor, and 4 dB shadow
231  // Retrieved from: https://mentor.ieee.org/802.11/dcn/14/11-14-0882-04-00ax-tgax-channel-model-document.docx
232  // IEEE 802.11ax uses the TGn channel D path loss model for performance evaluation of simulation scenario #2
233  // with extra indoor wall and floor penetration loss.
234  case PATH_LOSS_SCENARIO_2_TGax: {
235  int n_walls (2); // Wall frequency (n_walls walls each m)
236  int n_floors (3); // Floor frequency (n_floors floors each m)
237  int L_iw (7); // Penetration for a single wall (dB)
238  double LFS (32.4 + 20*log10(2.4*pow(10,3))+ 20*log10(distance/1000));
239  int d_BP (10); // Break-point distance (m)
240  if (distance >= d_BP) {
241  loss = LFS + 35*log10(distance/d_BP) + 18.3*pow(n_floors,((n_floors+2)/(n_floors+1)) - 0.46) + L_iw*n_walls;
242  } else {
243  loss = LFS;
244  }
245  pw_received_dbm = tx_power_dbm + tx_gain_db + rx_gain_db - loss;
246  pw_received = ConvertPower(DBM_TO_PW, pw_received_dbm);
247  break;
248  }
249 
250  // Indoor small BSSs
251  // Retrieved from: https://mentor.ieee.org/802.11/dcn/14/11-14-0882-04-00ax-tgax-channel-model-document.docx
252  // IEEE 802.11ax uses the TGn channel D path loss model for performance evaluation
253  // of simulation scenario #3.
254  case PATH_LOSS_SCENARIO_3_TGax: {
255  double LFS (32.4 + 20*log10(2.4*pow(10,3))+ 20*log10(distance/1000));
256  int d_BP (10); // Break-point distance (m)
257  if (distance >= d_BP) {
258  loss = LFS + 35*log10(distance/d_BP);
259  } else {
260  loss = LFS;
261  }
262  pw_received_dbm = tx_power_dbm + tx_gain_db + rx_gain_db - loss;
263  pw_received = ConvertPower(DBM_TO_PW, pw_received_dbm);
264  break;
265  }
266 
267  // Outdoor large BSS scenario
268  // Retrieved from: https://mentor.ieee.org/802.11/dcn/14/11-14-0882-04-00ax-tgax-channel-model-document.docx
269  case PATH_LOSS_SCENARIO_4_TGax: {
270  double h_AP (10); // Height of the AP in m
271  double h_STA (1.5); // Height of the STA in m
272  double d_BP ((4 * (h_AP - 1) * (h_STA - 1) * central_frequency) / SPEED_LIGHT);
273  if (distance < d_BP && distance >= 10) {
274  loss = 22 * log10(distance) + 28 + 20 *log10(central_frequency * pow(10,-9));
275  } else if (distance >= d_BP && distance < 5000) {
276  loss = 40 * log10(distance) + 7.8 + 18 * log10(h_AP - 1) -
277  18 * log10(h_STA - 1) + 20 * log10(central_frequency * pow(10,-9));
278  }
279  pw_received_dbm = tx_power_dbm + tx_gain_db - loss; // Power in dBm
280  pw_received = ConvertPower(DBM_TO_PW, pw_received_dbm);
281  break;
282  }
283 
284  // Outdoor large BSS scenario + Residential
285  // Retrieved from: https://mentor.ieee.org/802.11/dcn/14/11-14-0882-04-00ax-tgax-channel-model-document.docx
286  case PATH_LOSS_SCENARIO_4a_TGax: {
287  // double h_AP = 10; // Height of the AP in m
288  // double h_STA = 1.5; // Height of the STA in m
289  if (distance < 2000 && distance >= 10) {
290  loss = 36.7 * log10(distance) + 22.7 + 26 * log10(central_frequency * pow(10,-9));
291  }
292  // Outdoor-to-Indoor building penetration loss
293  // TODO: important to consider specifying d_outdoor and d_indoor
294  double d_outdoor (0);
295  double d_indoor (0);
296  loss = loss * (d_outdoor + d_indoor) + 20 + 0.5 * d_indoor;
297  pw_received_dbm = tx_power_dbm + tx_gain_db - loss; // Power in dBm
298  pw_received = ConvertPower(DBM_TO_PW, pw_received_dbm);
299  break;
300  }
301 
302  /*
303  * Medbo, J., & Berg, J. E. (2000). Simple and accurate path loss modeling at 5 GHz in indoor environments
304  * with corridors. In Vehicular Technology Conference, 2000. IEEE-VTS Fall VTC 2000. 52nd (Vol. 1, pp. 30-36). IEEE.
305  */
306  case PATHLOSS_5GHZ_OFFICE_BUILDING:{
307  // pl_overall = pl_free_space(d) + alpha * d
308  double pl_overall_db; // Overall path loss
309  double pl_free_space_db; // Pathloss free space
310  double alpha (0.44); // Constant attenuation per unit of path length [dB/m]
311  pl_free_space_db = 20 * log10(distance) + 20 * log10(central_frequency) +
312  20 * log10((4*M_PI)/((double) SPEED_LIGHT)) - ConvertPower(LINEAR_TO_DB, rx_gain) -
313  ConvertPower(LINEAR_TO_DB, tx_gain);
314  pl_overall_db = pl_free_space_db + alpha * distance;
315  double pw_received_dbm (ConvertPower(PW_TO_DBM, tx_power) - pl_overall_db);
316  pw_received = ConvertPower(DBM_TO_PW, pw_received_dbm);
317  break;
318  }
319 
320  /*
321  * Xu et al. Indoor Office Propagation Measurements and Path Loss Models at 5.25 GHz‚Äú, IEEE VTC 2007.
322  * one-slope log-distance model in in-room LoS condition
323  */
324  case PATHLOSS_INROOM_LOSS_5250KHZ:{
325  double pl_overall_db (47.8 + 14.8 * log10(distance)); // Overall path loss
326  double pw_received_dbm (ConvertPower(PW_TO_DBM, tx_power) - pl_overall_db);
327  pw_received = ConvertPower(DBM_TO_PW, pw_received_dbm);
328  break;
329  }
330 
331  /*
332  * Xu et al. Indoor Office Propagation Measurements and Path Loss Models at 5.25 GHz‚Äú, IEEE VTC 2007.
333  * dual-slope log-distance model in room-corridor condition
334  */
335  case PATHLOSS_ROOM_CORRIDOR_5250KHZ:{
336  double pl_overall_db;
337  if(distance <= 9){
338  pl_overall_db = 53.2 + 25.8 * log10(distance); // Overall path loss
339  } else {
340  pl_overall_db = 56.4 + 29.1 * log10(distance); // Overall path loss
341  }
342  double pw_received_dbm (ConvertPower(PW_TO_DBM, tx_power) - pl_overall_db);
343  pw_received = ConvertPower(DBM_TO_PW, pw_received_dbm);
344  break;
345  }
346 
347  default:{
348  printf("Path loss model not found!\n");
349  break;
350  }
351 
352  case PATHLOSS_TMB:{
353 
354 // L0 = 54.12;
355 // gamma = 2.06067;
356 // k = 5.25;
357 // W = 0.1467;
358 // loss = L0 + 10*gamma*log10(distance)+k*W*distance;
359 
360  double pl_overall_db;
361 
362  pl_overall_db = 54.12 + 10 * 2.06067 * log10(distance) + 5.25 * 0.1467 * distance;
363 
364  double pw_received_dbm = ConvertPower(PW_TO_DBM, tx_power) - pl_overall_db;
365 
366  pw_received = ConvertPower(DBM_TO_PW, pw_received_dbm);
367 
368  break;
369  }
370 
371  }
372 
373  return pw_received;
374 
375 }

407 void GetChannelOccupancyByCCA(int primary_channel, int pifs_activated, int *channels_free, int min_channel_allowed,
408  int max_channel_allowed, double **channel_power, double pd, double *timestampt_channel_becomes_free,
409  double sim_time, double pifs){
410 
411  switch(pifs_activated){
412 
413  case TRUE:{
414  double time_channel_has_been_free; // Time channel has been free since last P(ch) > CCA
415  for(int c = min_channel_allowed; c <= max_channel_allowed; ++c){
416  if(c == primary_channel){
417  if((*channel_power)[c] < pd) channels_free[c] = CHANNEL_FREE;

419  time_channel_has_been_free = sim_time - timestampt_channel_becomes_free[c];

424  channels_free[c] = CHANNEL_FREE;
425  } else {
426  channels_free[c] = CHANNEL_OCCUPIED;
427 
428  }
429  }
430  }
431  break;
432  }
433 
434  case FALSE:{
435  for(int c = min_channel_allowed; c <= max_channel_allowed; ++c){
436  if((*channel_power)[c] < pd){
437  channels_free[c] = CHANNEL_FREE;
438  } else {
439  channels_free[c] = CHANNEL_OCCUPIED;
440  }
441  }
442  break;
443  }

503 void ApplyAdjacentChannelInterferenceModel(int adjacent_channel_model, double total_power[],
504  Notification notification, int num_channels_komondor, double rx_gain,
505  double central_frequency, double pw_received, int path_loss_model){
506 
507  // Direct power (power of the channels used for transmitting)
508  for(int i = notification.left_channel; i <= notification.right_channel; ++i){
509  (total_power)[i] = pw_received;
510  }
511 
512  double pw_loss_db;
513  double total_power_dbm;
514 
515  // Co-channel interference power
516  switch(adjacent_channel_model){
517 
518  case ADJACENT_CHANNEL_NONE:{
519  // Do nothing
520  break;
521  }
522 
523  // (RECOMMENDED) Boundary co-channel interference: only boundary channels (left and right) used in the TX affect the rest of channels
524  case ADJACENT_CHANNEL_BOUNDARY:{
525  for(int c = 0; c < num_channels_komondor; ++c) {
526 
527  if(c < notification.left_channel || c > notification.right_channel){
528 
529  if(c < notification.left_channel) {
530 
531  pw_loss_db = 20 * abs(c-notification.left_channel);
532  total_power_dbm = ConvertPower(PW_TO_DBM, pw_received) - pw_loss_db;
533  (total_power)[c] = (total_power)[c] + ConvertPower(DBM_TO_PW, total_power_dbm);
534 
535  } else if(c > notification.right_channel) {
536 
537  pw_loss_db = 20 * abs(c-notification.right_channel);
538  total_power_dbm = ConvertPower(PW_TO_DBM, pw_received) - pw_loss_db;
539  (total_power)[c] = (total_power)[c] + ConvertPower(DBM_TO_PW, total_power_dbm);
540 
541  }
542 
543  if((total_power)[c] < MIN_VALUE_C_LANGUAGE){
544 
545  (total_power)[c] = 0;
546 
547  }
548 
549  } else {
550  // Inside TX range --> do nothing
551  }
552  }
553  break;
554  }
555 
556  case ADJACENT_CHANNEL_EXTREME:{
557 
558  for(int c = 0; c < num_channels_komondor; ++c) {
559 
560  for(int j = notification.left_channel; j <= notification.right_channel; ++j){
561 
562  if(c != j) {
563 
564  pw_loss_db = 20 * abs(c-j);
565  total_power_dbm = ConvertPower(PW_TO_DBM, pw_received) - pw_loss_db;
566  (total_power)[c] = (total_power)[c] + ConvertPower(DBM_TO_PW, total_power_dbm);
567  if((total_power)[c] < MIN_DOUBLE_VALUE_KOMONDOR) (total_power)[c] = 0;
568 
569  }
570  }
571  }
572  break;
573  }
574 
575  default:{
576  printf("ERROR: Unkown cochannel model!");
577  exit(EXIT_FAILURE);
578  break;
579  }
580  }
581 
582 }

654 void ComputeMaxInterference(double *max_pw_interference, int *channel_max_intereference,
655  Notification notification_interest, int node_state, std::map<int,double> &power_received_per_node,
