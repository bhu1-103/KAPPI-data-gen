92 void ComputeSimulationStatistics(Performance *performance_report, Configuration *configuration_per_node,
93  int total_nodes_number, int total_wlans_number){
94 
95  for(int m=0; m < total_nodes_number; ++m){
96  // Take into account only APs (transmitters)
97  if( configuration_per_node[m].capabilities.node_type == NODE_TYPE_AP ){
98  // Data packets sent in total
99  total_data_packets_sent = total_data_packets_sent + performance_report[m].data_packets_sent;
100  // Aggregate throughput
101  total_throughput = total_throughput + performance_report[m].throughput;
102  // Total number of packets generated
103  total_num_packets_generated = total_num_packets_generated + performance_report[m].num_packets_generated;
104  // Total number of RTS lost due to slotted backoff collisions
105  total_rts_lost_slotted_bo = total_rts_lost_slotted_bo + performance_report[m].rts_lost_slotted_bo;
106  // Total number of RTS sent
107  total_rts_cts_sent = total_rts_cts_sent + performance_report[m].rts_cts_sent;
108  // Sum of probabilities of finding a collision by slotted backoff
109  total_prob_slotted_bo_collision = total_prob_slotted_bo_collision + performance_report[m].prob_slotted_bo_collision;
110  // Total number of trials for transmitting
111  total_num_tx_init_not_possible = total_num_tx_init_not_possible + performance_report[m].num_tx_init_not_possible;
112  // Proportional fairness metric
113  proportional_fairness = proportional_fairness + log10(performance_report[m].throughput);
114  // Jain's fairness metric
115  jains_fairness_aux = jains_fairness_aux + pow(performance_report[m].throughput, 2);
116  // Sum of the average delay experienced by each transmitter
117  total_delay = total_delay + performance_report[m].average_delay;
118  // Maximum average delay in the network
119  if(performance_report[m].average_delay > max_delay) max_delay = performance_report[m].average_delay;
120  // Minimum average delay in the network
121  if(performance_report[m].average_delay < min_delay) min_delay = performance_report[m].average_delay;
122  // Average expected backoff
123  av_expected_backoff = av_expected_backoff + performance_report[m].expected_backoff;
124  // Average waiting time before transmitting
125  av_expected_waiting_time = av_expected_waiting_time + performance_report[m].average_waiting_time;
126  // Total bandwidth used in transmissions
127  total_bandiwdth_tx = total_bandiwdth_tx + performance_report[m].bandwidth_used_txing;
128  // Minimum throughput experienced in the network
129  if(performance_report[m].throughput < min_throughput) {
130  ix_wlan_min_throughput = m;
131  min_throughput = performance_report[m].throughput;
132  }
133  // Maximum throughput experienced in the network
134  if(performance_report[m].throughput > max_throughput) max_throughput = performance_report[m].throughput;
135  }
136  }
137  av_expected_backoff = av_expected_backoff / total_wlans_number;
138  av_expected_waiting_time = av_expected_waiting_time / total_wlans_number;
139  jains_fairness = pow(total_throughput, 2) / (total_nodes_number/2 * jains_fairness_aux); // Supposing that number_aps = number_nodes/2
140 }

155 void PrintAndWriteSimulationStatistics(int print_system_logs, int save_system_logs, Logger &logger_simulation,
156  Performance *performance_report, Configuration *configuration_per_node, int total_nodes_number,
157  int total_wlans_number, double frame_length, int max_num_packets_aggregated, double simulation_time_komondor) {
158 
159  // Compute global statistics
160  ComputeSimulationStatistics(performance_report, configuration_per_node, total_nodes_number, total_wlans_number);
161 
162  // Print final statistics in console logs
163  if (print_system_logs) {
164  printf("\n%s General Statistics (NEW FUNCTION):\n", LOG_LVL1);
165  printf("%s Average throughput per WLAN = %.3f Mbps (%.2f pkt/s)\n",
166  LOG_LVL2, (total_throughput * pow(10,-6)/total_wlans_number),
167  (total_throughput / (double) frame_length) /total_wlans_number);
168  printf("%s Min. throughput = %.2f Mbps (%.2f pkt/s)\n",
169  LOG_LVL3, min_throughput * pow(10,-6), min_throughput / (frame_length * max_num_packets_aggregated));
170  printf("%s Max. throughput = %.2f Mbps (%.2f pkt/s)\n",
171  LOG_LVL3, max_throughput * pow(10,-6), max_throughput / (frame_length * max_num_packets_aggregated));
172  printf("%s Total throughput = %.2f Mbps\n", LOG_LVL3, total_throughput * pow(10,-6));
173  printf("%s Total number of packets sent = %d\n", LOG_LVL3, total_data_packets_sent);
174  printf("%s Average number of data packets successfully sent per WLAN = %.2f\n",
175  LOG_LVL4, ((double) total_data_packets_sent/ (double) total_wlans_number));
176  printf("%s Average number of RTS packets lost due to slotted BO = %f (%.3f %% loss)\n",
177  LOG_LVL4, (double) total_rts_lost_slotted_bo/(double) total_wlans_number,
178  ((double) total_rts_lost_slotted_bo *100/ (double) total_rts_cts_sent));
179  printf("%s Average number of packets sent per WLAN = %d\n", LOG_LVL3, (total_data_packets_sent/total_wlans_number));
180  printf("%s Proportional Fairness = %.2f\n", LOG_LVL2, proportional_fairness);
181  printf("%s Jain's Fairness = %.2f\n", LOG_LVL2, jains_fairness);
182  printf("%s Prob. collision by slotted BO = %.3f\n", LOG_LVL2, total_prob_slotted_bo_collision / total_wlans_number);
183  printf("%s Av. delay = %.2f ms\n", LOG_LVL2, total_delay * pow(10,3) / total_wlans_number);
184  printf("%s Max. delay = %.2f ms\n", LOG_LVL3, max_delay * pow(10,3));
185  printf("%s Av. expected waiting time = %.2f ms\n", LOG_LVL3, av_expected_waiting_time * pow(10,3));
186  printf("%s Average bandwidth used for transmitting = %.2f MHz\n",
187  LOG_LVL2, total_bandiwdth_tx / (double) total_wlans_number);
188  printf("%s Time channel was idle = %.2f s (%f%%)\n", LOG_LVL2,
189  performance_report[0].sum_time_channel_idle, (100*performance_report[0].sum_time_channel_idle/simulation_time_komondor));
190  printf("\n\n");
191  }
192 
193  printf("\n");
194 
195  // Write final statistics in the file containing the console logs
196  if (save_system_logs) {
197  // Simulation log file
198  fprintf(logger_simulation.file,"\n%s General Statistics (NEW FUNCTION):\n", LOG_LVL1);
199  fprintf(logger_simulation.file,"%s Average throughput per WLAN = %.2f Mbps\n", LOG_LVL2, (total_throughput * pow(10,-6)/total_wlans_number));
200  fprintf(logger_simulation.file,"%s Total throughput = %.2f Mbps\n", LOG_LVL3, total_throughput * pow(10,-6));
201  fprintf(logger_simulation.file,"%s Total number of packets sent = %d\n", LOG_LVL3, total_data_packets_sent);
202  fprintf(logger_simulation.file,"%s Average number of data packets successfully sent per WLAN = %.2f\n",
203  LOG_LVL4, ( (double) total_data_packets_sent/ (double) total_wlans_number));
204  fprintf(logger_simulation.file,"%s Average number of RTS packets lost due to slotted BO = %.2f (%.2f %% loss)\n",
205  LOG_LVL4,
206  (double) total_rts_lost_slotted_bo/(double) total_wlans_number,
207  ((double) total_rts_lost_slotted_bo *100/ (double) total_rts_cts_sent));
208  fprintf(logger_simulation.file,"%s Average number of packets sent per WLAN = %d\n", LOG_LVL3, (total_data_packets_sent/total_wlans_number));
209  fprintf(logger_simulation.file,"%s Proportional Fairness = %.2f\n", LOG_LVL2, proportional_fairness);
210  fprintf(logger_simulation.file,"%s Jain's Fairness = %.2f\n", LOG_LVL2, jains_fairness);
211  fprintf(logger_simulation.file,"\n");
212  }
