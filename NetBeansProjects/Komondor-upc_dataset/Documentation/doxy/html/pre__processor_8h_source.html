157  double GenerateReward(int type_of_reward, Performance performance) {
158  double reward;
159  // Switch to select the reward according to the metric used (rewards must be normalized)
160  switch(type_of_reward){
161  /* PERFORMANCE_PACKETS_SENT:
162  * - The number of packets sent are taken into account
163  * - The reward must be bounded by the maximum number of data packets
164  * that can be sent in each interval (e.g., packets that were sent but lost)
165  */
166  case REWARD_TYPE_PACKETS_SENT:{
167  reward = performance.data_packets_sent/performance.data_packets_lost;
168  break;
169  }
170  /* PERFORMANCE_THROUGHPUT:
171  * - The throughput experienced during the last period is taken into account
172  * - The reward must be bounded by the maximum throughput that would be experienced
173  * (e.g., consider the data rate granted by the modulation and the total time)
174  */
175  case REWARD_TYPE_THROUGHPUT:{
176  if (performance.max_bound_throughput == 0) {
177  reward = 0;
178  } else {
179  reward = (double) performance.throughput/performance.max_bound_throughput;
180  }
181  break;
182  }
183  /* REWARD_TYPE_PACKETS_GENERATED:
184  * -
185  */
186  case REWARD_TYPE_PACKETS_GENERATED:{
187  reward = (performance.num_packets_generated - performance.num_packets_dropped) /
188  performance.num_packets_generated;
189  break;
190  }
191  /* Default */
192  default:{
193  printf("[Pre-Processor] ERROR: '%d' is not a correct type of performance indicator\n", type_of_reward);
194  PrintAvailableRewardTypes();
195  exit(EXIT_FAILURE);
196  break;
197  }
198  }
199  return reward;
200  }

239  Configuration GenerateNewConfigurationBandits(Configuration configuration, int ml_output){
240  // Find which parameters correspond to the selected arm
241  index2values(indexes_selected_arm, ml_output, num_actions_channel,
242  num_actions_sensitivity, num_actions_tx_power, num_actions_dcb_policy);
243  // Update each parameter according to the configuration provided by the MAB
244  int new_primary = list_of_channels[indexes_selected_arm[0]];
245  double new_pd = list_of_pd_values[indexes_selected_arm[1]];
246  double new_tx_power = list_of_tx_power_values[indexes_selected_arm[2]];
247  int new_dcb_policy = list_of_dcb_policy[indexes_selected_arm[3]];
248  // Generate the configuration object
249  Configuration new_configuration;
250  // Set configuration to the received one, and then change specific parameters
251  new_configuration = configuration;

253  new_configuration.selected_primary_channel = new_primary; // Primary
254  new_configuration.selected_pd = new_pd; // pd
255  new_configuration.selected_tx_power = new_tx_power; // TX Power
256  new_configuration.selected_dcb_policy = new_dcb_policy; // DCB policy
257  return new_configuration;

266  int FindActionIndexFromConfigurationBandits(Configuration configuration, int* &indexes_selected_arm) {
267  // Find the index of each chosen parameter
268  int index_channel = -1;
269  int index_pd = -1;
270  int index_tx_power = -1;
271  int index_dcb_policy = -1;
272  // Channel
273  for(int i = 0; i < num_actions_channel; i++) {
274  if(configuration.selected_primary_channel == list_of_channels[i]) {
275  index_channel = i;
276  }
277  }
278  // Packet Detection (PD) threshold
279  for(int i = 0; i < num_actions_sensitivity; i++) {
280  if(configuration.selected_pd == list_of_pd_values[i]) {
281  index_pd = i;
282  }
283  }
284  // Tx Power
285  for(int i = 0; i < num_actions_tx_power; i++) {
286  if(configuration.selected_tx_power == list_of_tx_power_values[i]) {
287  index_tx_power = i;
288  }
289  }
290  // DCB policy
291  for(int i = 0; i < num_actions_dcb_policy; i++) {
292  if(configuration.selected_dcb_policy == list_of_dcb_policy[i]) {
293  index_dcb_policy = i;
294  }
295  }
296  // Update the index of each chosen parameter
297  indexes_selected_arm[0] = index_channel;
298  indexes_selected_arm[1] = index_pd;
299  indexes_selected_arm[2] = index_tx_power;
300  indexes_selected_arm[3] = index_dcb_policy;
301  // Find the action ix and return it
302  int action_ix = values2index(indexes_selected_arm, num_actions_channel,
303  num_actions_sensitivity, num_actions_tx_power, num_actions_dcb_policy);
304  return action_ix;
305  }

362  void index2values(int *indexes, int action_ix, int size_channels, int size_pd, int size_tx_power, int size_dcb_policy) {
363  indexes[0] = (int) action_ix/(size_pd * size_tx_power * size_dcb_policy);
364  indexes[1] = (int) (action_ix - indexes[0] * (size_pd * size_tx_power * size_dcb_policy))
365  /(size_tx_power * size_dcb_policy);
366  indexes[2] = (int) (action_ix - indexes[0] * (size_pd * size_tx_power * size_dcb_policy)
367  - indexes[1] * (size_tx_power * size_dcb_policy))
368  /(size_dcb_policy);
369  indexes[3] = action_ix % size_dcb_policy;
370  }

380  int values2index(int *indexes, int size_channels, int size_pd, int size_tx_power, int size_dcb_policy) {
381  int index = indexes[0] * (size_pd * size_tx_power * size_dcb_policy)
382  + indexes[1] * (size_tx_power * size_dcb_policy)
383  + indexes[2] * size_dcb_policy
384  + indexes[3];
385  return index;
386  }
