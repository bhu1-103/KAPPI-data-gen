164  void GraphColoringOptimization(Configuration *configuration_array) {
165 
166  // For each AP "i", find the maximum weight to AP "j" sharing the same channel
167  double max_weight;
168  bool flag_change_channel;
169  for (int i = 0; i < agents_number; ++ i) {
170  max_weight = -100000;
171  flag_change_channel = false;
172  for (int j = 0; j < agents_number; ++ j) {
173  if (i != j && configuration_array[i].selected_primary_channel ==
174  configuration_array[j].selected_primary_channel) {
175  flag_change_channel = true;
176  if (weight_edges[i][j] > max_weight) {
177  max_weight = weight_edges[i][j];
178  }
179  }
180  }
181  if (flag_change_channel) {
182  // Choose a new color for AP "i", so that H(c) = min H(c) for c = 1...k
183  configuration_array[i].selected_primary_channel = FindBestChannel(i, configuration_array[i].selected_primary_channel);
184  }
185 // printf("NEW CHANNEL (agent %d) = %d\n", i, configuration_array[i].selected_primary_channel);
186  }
187  }
