203 void Agent :: Start(){
204 
205  // Create agent logs file if required
206  if(save_agent_logs) {
207  // Name agent log file accordingly to the agent_id
208  sprintf(own_file_path,"%s_A%d_%s.txt","../output/logs_output", agent_id, wlan_code.c_str());
209  remove(own_file_path);
210  output_log_file = fopen(own_file_path, "at");
211  agent_logger.save_logs = save_agent_logs;
212  agent_logger.file = output_log_file;
213  agent_logger.SetVoidHeadString();
214  }
215 
216  LOGS(save_agent_logs, agent_logger.file,
217  "%.18f;A%d;%s;%s Start()\n", SimTime(), agent_id, LOG_B00, LOG_LVL1);
218 
219  if(communication_level == PURE_CENTRALIZED) {
220  // --- Do nothing ---
221  // In case of being centralized, wait for a request from the controller
222  printf("%s Agent %d: Learning operation managed by the CC\n", LOG_LVL3, agent_id);
223  } else {
224  // Initialize the PP and the ML Method
225  InitializePreProcessor();
226  InitializeMlModel();
227  // Generate the first request to be triggered after "time_between_requests"
228  // *** We generate here the first request in order to obtain the AP's configuration
229  double extra_wait_test = 0.005 * (double) agent_id;
230  trigger_request_information_to_ap.Set(FixTimeOffset(SimTime() + time_between_requests + extra_wait_test,13,12));
231  }
232 
233 };

281 void Agent :: InportReceivingInformationFromAp(Configuration &received_configuration, Performance &received_performance){
282 
283 // printf("%s Agent #%d: Message received from the AP\n", LOG_LVL1, agent_id);
284 
285  LOGS(save_agent_logs, agent_logger.file,
286  "%.15f;A%d;%s;%s InportReceivingInformationFromAp()\n",
287  SimTime(), agent_id, LOG_F00, LOG_LVL1);
288 
289  LOGS(save_agent_logs, agent_logger.file,
290  "%.15f;A%d;%s;%s New information has been received from the AP\n",
291  SimTime(), agent_id, LOG_C00, LOG_LVL2);
292 
293  configuration = received_configuration;
294  performance = received_performance;
295 
296  if(save_agent_logs) {
297  WriteConfiguration(configuration);
298  WritePerformance(performance);
299  }
300 
301  // Forward information to the CC, if required
302  if (communication_level == PURE_CENTRALIZED || communication_level == HYBRID_CENTRALIZED_DECENTRALIZED) {
303  // Forward the information to the controller
304  LOGS(save_agent_logs, agent_logger.file,
305  "%.15f;A%d;%s;%s Answering to the controller with current information\n",
306  SimTime(), agent_id, LOG_F02, LOG_LVL2);
307  outportAnswerToController(configuration, performance, agent_id);
308  }
309 
310  // Compute a new configuration in case of not using the pure centralized approach
311  if (communication_level != PURE_CENTRALIZED) {
312  // Compute a new configuration according to the updated rewards
313  ComputeNewConfiguration();
314  }
315

419 void Agent :: ComputeNewConfiguration(){
420 
421  LOGS(save_agent_logs, agent_logger.file, "%.15f;A%d;%s;%s ComputeNewConfiguration()\n",
422  SimTime(), agent_id, LOG_F00, LOG_LVL1);
423 
424  // Process the configuration and performance reports obtained from the WLAN
425  processed_configuration = pre_processor.ProcessWlanConfiguration(MULTI_ARMED_BANDITS, configuration);
426  processed_performance = pre_processor.ProcessWlanPerformance(MULTI_ARMED_BANDITS, performance, type_of_reward);
427 
428  // Update the configuration according to the selected learning method
429  switch(learning_mechanism) {
430 
431  /* Multi-Armed Bandits */
432  case MULTI_ARMED_BANDITS:{
433  // Update the configuration according to the MABs operation
434  ML_output = ml_model.ComputeIndividualConfiguration
435  (processed_configuration, processed_performance, agent_logger, SimTime());
436  break;
437  }
438  default:{
439  printf("[AGENT] ERROR: %d is not a correct learning mechanism\n", learning_mechanism);
440  ml_model.PrintAvailableLearningMechanisms();
441  exit(EXIT_FAILURE);
442  break;
443  }
444  }
445 
446  // Process the configuration from the output of the ML method
447  new_configuration = pre_processor.ProcessMLOutput(learning_mechanism, configuration, ML_output);
448 
449  // Send the configuration to the AP
450  SendNewConfigurationToAp(new_configuration);
451 
452 }
