69 double ComputeBackoff(int pdf_backoff, int cw, int backoff_type){
70 
71  double backoff_time;
72  double expected_backoff ((double) (cw-1)/2); // [slots]
73  double lambda_backoff (1/(expected_backoff * SLOT_TIME));
74 
75  switch(pdf_backoff){
76 
77  case PDF_DETERMINISTIC:{
78  if(backoff_type == BACKOFF_SLOTTED) {
79  int num_slots (rand() % cw); // Num slots in [0, CW-1]
80  backoff_time = num_slots * SLOT_TIME;
81  // printf("num_slots = %d\n", num_slots);
82  } else if(backoff_type == BACKOFF_CONTINUOUS) {
83  backoff_time = 1/lambda_backoff;
84  }
85  break;
86  }
87 
88  case PDF_EXPONENTIAL:{
89  if(backoff_type == BACKOFF_SLOTTED) {
90  backoff_time = round(Exponential2(expected_backoff)) * SLOT_TIME;
91  } else if(backoff_type == BACKOFF_CONTINUOUS) {
92  backoff_time = Exponential2(1/lambda_backoff);
93  }
94  break;
95  }
96 
97  default:{
98  printf("Backoff model not found!\n");
99  break;
100  }
101  }
102 
103  return backoff_time;
104 
105 }

113 double ComputeRemainingBackoff(int backoff_type, double remaining_backoff){
114 
115  double updated_remaining_backoff;
116 
117  switch(backoff_type){
118 
119  case BACKOFF_SLOTTED: {
120  int closest_slot (round(remaining_backoff / SLOT_TIME));
121 // printf("- closest_slot = %d\n", closest_slot);
122  if(fabs(remaining_backoff - closest_slot * SLOT_TIME) < MAX_DIFFERENCE_SAME_TIME){
123  updated_remaining_backoff = closest_slot * SLOT_TIME;
124  } else {
125  updated_remaining_backoff = ceil(remaining_backoff/SLOT_TIME) * SLOT_TIME;
126  }
127  break;
128  }
129 
130  case BACKOFF_CONTINUOUS: {
131  updated_remaining_backoff = remaining_backoff;
132  break;
133  }
134 
135  default:{
136  printf("Backoff type not found!\n");
137  exit(EXIT_FAILURE);
138  break;
139  }
140 
141  }
142 
143  return updated_remaining_backoff;
144 
145 }

156 int HandleBackoff(int pause_or_resume, double **channel_power, int primary_channel, double pd,
157  int packets_in_buffer){
158 
159  int backoff_action (FALSE);
160 
161  switch(pause_or_resume){
162 
163  case PAUSE_TIMER:{
164  if((*channel_power)[primary_channel] > pd) backoff_action = TRUE;
165  break;
166  }
167 
168  case RESUME_TIMER:{
169  if(packets_in_buffer > 0) {
170  if((*channel_power)[primary_channel] <= pd) backoff_action = TRUE;
171  }
172  break;
173  }
174 
175  default:{
176  printf("Unknown handle backoff mode! (not resume nor pause)\n");
177  exit(EXIT_FAILURE);
178  }
179  }
180 
181  return backoff_action;
182

194 void HandleContentionWindow(int cw_adaptation, int increase_or_reset, int* cw_current, int cw_min,
195  int *cw_stage_current, int cw_stage_max) {
196  if(cw_adaptation == TRUE){
197  switch(increase_or_reset) {
198  case INCREASE_CW:{
199  if(*cw_stage_current < cw_stage_max){
200  *cw_stage_current = *cw_stage_current + 1;
201  *cw_current = cw_min * pow(2, *cw_stage_current);
202  }
203  break;
204  }
205  case RESET_CW:{
206  *cw_stage_current = 0;
207  *cw_current = cw_min;
208  break;
209  }
210  default:{
211  printf("Unknown operation on contention window!");
212  exit(EXIT_FAILURE);
213  break;
214  }
215  }
216  } else {
217  // Constant CW - do nothing: keep cw
218  }
